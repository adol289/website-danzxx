Metode QR Code
  • Install dependencies 
    npm install @whiskeysockets/baileys qrcode-terminal pino sharp axios ytdl-core
  • Script QR dengan fitur lengkap 
    const makeWASocket = require('@whiskeysockets/baileys').default;
const { useMultiFileAuthState, DisconnectReason } = require('@whiskeysockets/baileys');
const QRCode = require('qrcode-terminal');
const Pino = require('pino');

async function startBot() {
    const { state, saveCreds } = await useMultiFileAuthState('auth_qr');
    const sock = makeWASocket({
        auth: state,
        logger: Pino({ level: 'silent' }),
        printQRInTerminal: true,
        browser: ['UltimateBot', 'Chrome', '120']
    });

    sock.ev.on('connection.update', (update) => {
        const { connection, qr } = update;
        if (qr) QRCode.generate(qr, { small: true });
        if (connection === 'open') console.log('✅ Bot Connected!');
    });

    sock.ev.on('creds.update', saveCreds);
    
    // Handler dengan semua fitur (lihat di tab Super Fitur)
    require('./handlers/messageHandler')(sock);
}

startBot();

// handlers/messageHandler.js
const fs = require('fs');
const axios = require('axios');
const ytdl = require('ytdl-core');
const sharp = require('sharp');
const moment = require('moment-timezone');
const { GoogleGenerativeAI } = require('@google/generative-ai');

const prefix = process.env.PREFIX || '!';
const ownerNumber = process.env.OWNER_NUMBER;
const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);

module.exports = (sock) => {
    sock.ev.on('messages.upsert', async ({ messages }) => {
        const msg = messages[0];
        if (!msg.message || msg.key.fromMe) return;
        
        const remoteJid = msg.key.remoteJid;
        const sender = msg.key.participant || msg.key.remoteJid;
        const text = msg.message.conversation || msg.message.extendedTextMessage?.text || '';
        if (!text.startsWith(prefix)) return;
        
        const args = text.slice(prefix.length).trim().split(' ');
        const command = args.shift().toLowerCase();
        
        // ============ STICKER MAKER DENGAN CUSTOM AUTHOR ============
        if (command === 'sticker') {
            let author = 'PutraBot';
            let pack = 'Putra Corporation';
            let stickerText = args.join(' ');
            
            if (stickerText.includes('|')) {
                const parts = stickerText.split('|');
                stickerText = parts[0].trim();
                if (parts[1]) author = parts[1].trim();
                if (parts[2]) pack = parts[2].trim();
            }
            
            // Sticker dari gambar yang dikirim
            if (msg.message.imageMessage || msg.message.videoMessage) {
                const media = await downloadMediaMessage(msg, 'buffer', {});
                await sock.sendMessage(remoteJid, {
                    sticker: media,
                    packname: pack,
                    author: author
                });
            } 
            // Sticker dari teks
            else if (stickerText) {
                const svg = ``;
                const buffer = await sharp(Buffer.from(svg)).png().toBuffer();
                await sock.sendMessage(remoteJid, {
                    sticker: buffer,
                    packname: pack,
                    author: author
                });
            }
        }
        
        // ============ STICKER KE GAMBAR ============
        if (command === 'toimg' && msg.message.stickerMessage) {
            const buffer = await downloadMediaMessage(msg, 'buffer', {});
            await sock.sendMessage(remoteJid, { image: buffer, caption: '✅ Sticker converted to image' });
        }
        
        // ============ AI CHATBOT (GEMINI) ============
        if (command === 'ai') {
            const prompt = args.join(' ');
            if (!prompt) {
                await sock.sendMessage(remoteJid, { text: '❌ Masukkan pertanyaan!\nContoh: !ai apa itu coding?' });
                return;
            }
            try {
                const model = genAI.getGenerativeModel({ model: 'gemini-pro' });
                const result = await model.generateContent(prompt);
                const response = result.response.text();
                await sock.sendMessage(remoteJid, { text: `🤖 *AI Response:*\n${response}` });
            } catch (err) {
                await sock.sendMessage(remoteJid, { text: '❌ AI Error: ' + err.message });
            }
        }
        
        // ============ DOWNLOAD YOUTUBE MP3 ============
        if (command === 'ytmp3') {
            const url = args[0];
            if (!url) {
                await sock.sendMessage(remoteJid, { text: '❌ Masukkan URL YouTube!\nContoh: !ytmp3 https://youtu.be/xxxxx' });
                return;
            }
            await sock.sendMessage(remoteJid, { text: '⏳ Sedang memproses audio...' });
            try {
                const info = await ytdl.getInfo(url);
                const audioStream = ytdl(url, { filter: 'audioonly', quality: 'highestaudio' });
                await sock.sendMessage(remoteJid, { 
                    audio: audioStream, 
                    mimetype: 'audio/mpeg',
                    fileName: `${info.videoDetails.title}.mp3`
                });
            } catch (err) {
                await sock.sendMessage(remoteJid, { text: '❌ Gagal download: ' + err.message });
            }
        }
        
        // ============ DOWNLOAD YOUTUBE MP4 ============
        if (command === 'ytmp4') {
            const url = args[0];
            if (!url) {
                await sock.sendMessage(remoteJid, { text: '❌ Masukkan URL YouTube!' });
                return;
            }
            await sock.sendMessage(remoteJid, { text: '⏳ Mengunduh video...' });
            try {
                const videoStream = ytdl(url, { filter: 'videoandaudio', quality: 'highest' });
                await sock.sendMessage(remoteJid, { video: videoStream, caption: '✅ Video downloaded!' });
            } catch (err) {
                await sock.sendMessage(remoteJid, { text: '❌ Error: ' + err.message });
            }
        }
        
        // ============ TAGALL ============
        if (command === 'tagall' && remoteJid.endsWith('@g.us')) {
            const groupMetadata = await sock.groupMetadata(remoteJid);
            const participants = groupMetadata.participants;
            let mentions = [];
            let messageText = '📢 *PENGUMUMAN*\n\n';
            participants.forEach(p => {
                mentions.push(p.id);
                messageText += `@${p.id.split('@')[0]}\n`;
            });
            await sock.sendMessage(remoteJid, { text: messageText, mentions });
        }
        
        // ============ WEATHER ============
        if (command === 'weather') {
            const city = args.join(' ');
            if (!city) {
                await sock.sendMessage(remoteJid, { text: '❌ Masukkan nama kota!\nContoh: !weather Jakarta' });
                return;
            }
            try {
                const apiKey = process.env.WEATHER_API_KEY || 'your_key';
                const response = await axios.get(`http://api.openweathermap.org/data/2.5/weather?q=${city}&appid=${apiKey}&units=metric&lang=id`);
                const data = response.data;
                const weatherText = `🌤️ *Cuaca di ${data.name}*\n\n🌡️ Suhu: ${data.main.temp}°C\n💧 Kelembaban: ${data.main.humidity}%\n🌬️ Angin: ${data.wind.speed} m/s\n📝 Kondisi: ${data.weather[0].description}`;
                await sock.sendMessage(remoteJid, { text: weatherText });
            } catch (err) {
                await sock.sendMessage(remoteJid, { text: '❌ Kota tidak ditemukan!' });
            }
        }
        
        // ============ JADWAL SHOLAT ============
        if (command === 'jadwalsholat') {
            const city = args.join(' ') || 'jakarta';
            try {
                const response = await axios.get(`https://api.myquran.com/v1/sholat/kota/cari/${city}`);
                const kotaId = response.data.data[0]?.id;
                if (!kotaId) throw new Error('Kota tidak ditemukan');
                const jadwal = await axios.get(`https://api.myquran.com/v1/sholat/jadwal/${kotaId}/${moment().format('YYYY/MM/DD')}`);
                const data = jadwal.data.data.jadwal;
                const text = `🕌 *Jadwal Sholat ${data.tanggal}*\n\n🌅 Imsak: ${data.imsak}\n🌄 Subuh: ${data.subuh}\n☀️ Dzuhur: ${data.dzuhur}\n🌇 Ashar: ${data.ashar}\n🌆 Maghrib: ${data.maghrib}\n🌙 Isya: ${data.isya}`;
                await sock.sendMessage(remoteJid, { text });
            } catch (err) {
                await sock.sendMessage(remoteJid, { text: '❌ Gagal mengambil jadwal sholat' });
            }
        }
        
        // ============ RANDOM QUOTES ============
        if (command === 'quotes') {
            const quotes = [
                'Hidup adalah tentang perjalanan, bukan tujuan.',
                'Kegagalan adalah batu loncatan menuju kesuksesan.',
                'Jangan menyerah, karena kamu tidak tahu seberapa dekat kamu dengan tujuan.',
                'Setiap hari adalah kesempatan baru untuk menjadi lebih baik.'
            ];
            const randomQuote = quotes[Math.floor(Math.random() * quotes.length)];
            await sock.sendMessage(remoteJid, { text: `💭 *Quote Hari Ini*\n\n${randomQuote}` });
        }
        
        // ============ RUNTIME ============
        if (command === 'runtime') {
            const uptime = process.uptime();
            const days = Math.floor(uptime / 86400);
            const hours = Math.floor((uptime % 86400) / 3600);
            const minutes = Math.floor((uptime % 3600) / 60);
            const seconds = Math.floor(uptime % 60);
            await sock.sendMessage(remoteJid, { text: `⏱️ *Bot Runtime*\n${days}d ${hours}h ${minutes}m ${seconds}s` });
        }
        
        // ============ PING ============
        if (command === 'ping') {
            const start = Date.now();
            await sock.sendMessage(remoteJid, { text: '🏓 Pinging...' });
            const end = Date.now();
            await sock.sendMessage(remoteJid, { text: `🏓 Pong! ${end - start}ms` });
        }
        
        // ============ MENU ============
        if (command === 'menu' || command === 'help') {
            const menu = `🤖 *${process.env.BOT_NAME || 'Ultimate Bot'}*\n━━━━━━━━━━━━━━━━━━\n📌 *Prefix:* ${prefix}\n\n✨ *STICKER*\n${prefix}sticker [text] | [author] | [pack]\n${prefix}toimg\n\n🤖 *AI*\n${prefix}ai [prompt]\n\n📥 *DOWNLOADER*\n${prefix}ytmp3 [url]\n${prefix}ytmp4 [url]\n\n👥 *GROUP*\n${prefix}tagall\n\n🌤️ *UTILITY*\n${prefix}weather [kota]\n${prefix}jadwalsholat [kota]\n${prefix}quotes\n${prefix}ping\n${prefix}runtime\n\n👑 *OWNER*\n${prefix}owner\n\n━━━━━━━━━━━━━━━━━━\n💡 Ketik ${prefix}help untuk melihat menu ini lagi`;
            await sock.sendMessage(remoteJid, { text: menu });
        }
        
        // ============ OWNER INFO ============
        if (command === 'owner') {
            await sock.sendMessage(remoteJid, { text: `👑 *Owner Bot*\nNomor: ${ownerNumber}\nBot: ${process.env.BOT_NAME}\n💻 Dibuat dengan Baileys` });
        }
    });
};

// Helper function
async function downloadMediaMessage(msg, type, opts) {
    const { downloadMediaMessage } = require('@whiskeysockets/baileys');
    return await downloadMediaMessage(msg, type, opts);
}
