Baileys Bot Installer

Panduan profesional instalasi library Baileys — pairing code atau QR Code — lengkap dengan fitur sticker, image, dan command handler

Node.js v18+ Baileys@6.x QR / Pairing Sticker + Author
Metode Pairing Code (Pairing Code)

Metode pairing memungkinkan bot terhubung hanya dengan memasukkan nomor WhatsApp tanpa memindai QR. Rekomendasi untuk production.

  • Buat folder project & installasi awal 
    mkdir baileys-bot && cd baileys-bot
npm init -y
  • Install dependencies utama 
    npm install @whiskeysockets/baileys@6.5.0 @adiwajshing/keyed-db pino qrcode-terminal dotenv
npm install sharp file-type axios @ffmpeg-installer/ffmpeg fluent-ffmpeg

    NOMOR_PAIRING=6281234567890
PAIRING_CODE=true
OWNER_NUMBER=6281234567890

    const makeWASocket = require('@whiskeysockets/baileys').default;
const { useMultiFileAuthState, DisconnectReason, downloadMediaMessage } = require('@whiskeysockets/baileys');
const { Boom } = require('@whiskeysockets/baileys');
const Pino = require('pino');
const fs = require('fs');
const sharp = require('sharp');
const axios = require('axios');
require('dotenv').config();

// Konfigurasi sticker default
const defaultStickerAuthor = process.env.BOT_NAME || 'PutraBot';
const defaultStickerPack = 'Putra Corporation';

async function startBot() {
    const { state, saveCreds } = await useMultiFileAuthState('auth_info');
    const sock = makeWASocket({
        auth: state,
        logger: Pino({ level: 'silent' }),
        printQRInTerminal: false,
        browser: ['PutraBot', 'Chrome', '120.0.0']
    });

    sock.ev.on('connection.update', async (update) => {
        const { connection, lastDisconnect } = update;
        if (connection === 'open') {
            console.log('✅ Bot berhasil terhubung!');
        }
        if (connection === 'close') {
            const shouldReconnect = (lastDisconnect.error?.output?.statusCode !== DisconnectReason.loggedOut);
            if (shouldReconnect) startBot();
            else console.log('❌ Session logout, hapus folder auth_info dan ulang.');
        }
    });

    sock.ev.on('creds.update', saveCreds);

    // Handler pesan dengan fitur lengkap
    sock.ev.on('messages.upsert', async ({ messages }) => {
        const msg = messages[0];
        if (!msg.message || msg.key.fromMe) return;
        
        const remoteJid = msg.key.remoteJid;
        const text = msg.message.conversation || msg.message.extendedTextMessage?.text || '';
        const lowerText = text.toLowerCase();
        
        // ============ STICKER MAKER ============
        // Sticker dari teks: !sticker teks|author|pack
        if (lowerText.startsWith('!sticker')) {
            const args = text.slice(9).trim();
            let stickerText = args;
            let author = defaultStickerAuthor;
            let pack = defaultStickerPack;
            
            if (args.includes('|')) {
                const parts = args.split('|');
                stickerText = parts[0].trim();
                if (parts[1]) author = parts[1].trim();
                if (parts[2]) pack = parts[2].trim();
            }
            
            if (!stickerText) {
                await sock.sendMessage(remoteJid, { text: '📝 *Cara pakai:*\n!sticker [teks] | [author] | [pack]\n\nContoh: !sticker Putra Corp | Xyz | PutraBot' });
                return;
            }
            
            try {
                // Generate image dari teks menggunakan canvas-like via sharp (simulasi)
                const svg = ``;
                const buffer = await sharp(Buffer.from(svg)).png().toBuffer();
                
                await sock.sendMessage(remoteJid, {
                    sticker: buffer,
                    mimetype: 'image/png',
                    packname: pack,
                    author: author
                });
            } catch (err) {
                await sock.sendMessage(remoteJid, { text: '❌ Gagal membuat sticker.' });
            }
        }
        
        // Sticker dari gambar/gif yang dikirim dengan caption !sticker author|pack
        if (msg.message.imageMessage || msg.message.videoMessage) {
            const caption = msg.message.imageMessage?.caption || msg.message.videoMessage?.caption || '';
            if (caption.toLowerCase().startsWith('!sticker')) {
                let author = defaultStickerAuthor;
                let pack = defaultStickerPack;
                const args = caption.slice(9).trim();
                if (args.includes('|')) {
                    const parts = args.split('|');
                    if (parts[0]) author = parts[0].trim();
                    if (parts[1]) pack = parts[1].trim();
                }
                
                const media = await downloadMediaMessage(msg, 'buffer', {}, { logger: Pino({ level: 'silent' }), reuploadRequest: sock.updateMediaMessage });
                await sock.sendMessage(remoteJid, {
                    sticker: media,
                    mimetype: msg.message.imageMessage ? 'image/png' : 'video/mp4',
                    packname: pack,
                    author: author
                });
            }
        }
        
        // ============ IMAGE PROCESSING ============
        if (lowerText === '!toimg' && msg.message.stickerMessage) {
            const stickerBuffer = await downloadMediaMessage(msg, 'buffer', {}, { logger: Pino({ level: 'silent' }) });
            await sock.sendMessage(remoteJid, { image: stickerBuffer, caption: '✅ Konversi sticker ke gambar berhasil' });
        }
        
        // ============ COMMAND LIST ============
        if (lowerText === '!menu' || lowerText === '!help') {
            const menu = `🤖 *${defaultStickerPack} Bot Menu*
            
✨ *Sticker*
!sticker [teks] | [author] | [pack]
Kirim gambar/video dengan caption !sticker author|pack

🖼️ *Image*
!toimg - convert sticker ke gambar

📥 *Downloader*
!ytmp3 [url] - download audio YouTube
!ytmp4 [url] - download video YouTube

👤 *Info*
!owner - info pemilik bot
!ping - cek status bot

📌 *Contoh sticker custom:*
!sticker Hello World | Xyz | MyPack`;
            
            await sock.sendMessage(remoteJid, { text: menu });
        }
        
        if (lowerText === '!ping') {
            await sock.sendMessage(remoteJid, { text: '🏓 Pong! Bot aktif dan berfungsi dengan baik.' });
        }
        
        if (lowerText === '!owner') {
            await sock.sendMessage(remoteJid, { text: `👑 *Owner Bot*\nNomor: ${process.env.OWNER_NUMBER || '6281234567890'}\nBot: ${defaultStickerPack}` });
        }
        
        // ============ DOWNLOADER SEDERHANA (YOUTUBE) ============
        if (lowerText.startsWith('!ytmp3') || lowerText.startsWith('!ytmp4')) {
            const url = text.split(' ')[1];
            if (!url) {
                await sock.sendMessage(remoteJid, { text: '❌ Masukkan URL YouTube!\nContoh: !ytmp3 https://youtu.be/xxxxx' });
                return;
            }
            await sock.sendMessage(remoteJid, { text: '⏳ Fitur downloader membutuhkan instalasi ytdl-core. Install dengan:\nnpm install ytdl-core\n\nSedang dalam pengembangan lebih lanjut.' });
        }
    });
    
    // Request Pairing Code
    const nomorTarget = process.env.NOMOR_PAIRING;
    if (nomorTarget) {
        setTimeout(async () => {
            const code = await sock.requestPairingCode(nomorTarget);
            console.log(`✨ Kode Pairing untuk ${nomorTarget}: ${code}`);
            console.log(`Masukkan kode tersebut di WhatsApp perangkat utama (LINK DEVICE) ✅`);
        }, 3000);
    }
}

  • Buat script qr-bot.js 
    const makeWASocket = require('@whiskeysockets/baileys').default;
const { useMultiFileAuthState, DisconnectReason, downloadMediaMessage } = require('@whiskeysockets/baileys');
const Pino = require('pino');
const QRCode = require('qrcode-terminal');
const sharp = require('sharp');
require('dotenv').config();

async function startBot() {
    const { state, saveCreds } = await useMultiFileAuthState('auth_qr');
    const sock = makeWASocket({
        auth: state,
        logger: Pino({ level: 'silent' }),
        printQRInTerminal: true,
        browser: ['PutraBotQR', 'Edge', '120']
    });

    sock.ev.on('connection.update', (update) => {
        const { connection, lastDisconnect, qr } = update;
        if (qr) {
            QRCode.generate(qr, { small: true });
            console.log('📱 Scan QR Code di atas menggunakan WhatsApp');
        }
        if (connection === 'open') {
            console.log('✅ Bot berhasil terhubung!');
        }
        if (connection === 'close') {
            const code = lastDisconnect.error?.output?.statusCode;
            if (code !== DisconnectReason.loggedOut) startBot();
            else console.log('Session expired, hapus folder auth_qr');
        }
    });

    sock.ev.on('creds.update', saveCreds);
    
    // Fitur sama seperti pairing, bisa copy paste handler dari pairing
    sock.ev.on('messages.upsert', async ({ messages }) => {
        const msg = messages[0];
        if (!msg.message || msg.key.fromMe) return;
        const remoteJid = msg.key.remoteJid;
        const text = msg.message.conversation || msg.message.extendedTextMessage?.text || '';
        
        if (text.toLowerCase() === '!menu') {
            await sock.sendMessage(remoteJid, { text: '🤖 Bot aktif dengan QR Method! Gunakan !sticker, !toimg, !ping' });
        }
        if (text.toLowerCase() === '!ping') {
            await sock.sendMessage(remoteJid, { text: '🏓 Pong!' });
        }
    });
}

startBot();

Fitur Lengkap Bot WhatsApp

Sticker Maker

Buat sticker dari teks, gambar, atau GIF dengan custom author & pack name.

!sticker [teks] | [author] | [pack] Kirim gambar + caption !sticker author|pack
Contoh: !sticker Putra Corp | Xyz | BotPack

Image Processing

Konversi sticker ke gambar, resize, dan manipulasi gambar lainnya.

!toimg !resize width height

Downloader

Download audio/video dari YouTube, Instagram, TikTok.

!ytmp3 [url] !ytmp4 [url] !ig [url]

Auto Reply & AI

Fitur auto-reply custom, chat AI (Gemini/OpenAI), dan welcome message.

!setwelcome [on/off] !ai [prompt]

Group Management

Kick, add, promote, demote, dan setting group otomatis.

!kick @user !promote @user !tagall

Utility

Cek ping, info bot, owner, dan status server.

!menu / !help !ping !owner !runtime
Custom Sticker dengan Nama & Author Request: 
// Format yang didukung
!sticker Hello World | Xyz | PutraCorp
// Hasil: sticker dengan teks "Hello World", author "Xyz", pack "PutraCorp"

// Kirim gambar dengan caption:
!sticker | MyAuthor | MyPack
// Sticker akan menggunakan gambar yang dikirim

⚠️ Troubleshooting Fitur:

  • Sticker tidak muncul: Pastikan install sharp dan format media didukung.
  • Downloader error: Install ytdl-core dan ffmpeg terlebih dahulu.
  • Custom author/pack tidak berfungsi: Gunakan format !sticker teks | author | pack dengan tanda pipa (|).
  • Session expired: Hapus folder auth_* dan pairing ulang.
